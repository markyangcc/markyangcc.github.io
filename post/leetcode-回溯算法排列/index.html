class Solution {
private:
    vector<int> path;//全局变量，在permute()函数内部定义并传递引用也可
    vector<vector<int>> res;

public:
    vector<vector<int>> permute(vector<int>& nums) {
        vector<bool> used(nums.size(),false);//排列问题需要used数组,组合问题需要startindex
        backtracking(nums,used);
        return res;
    }

    void backtracking(vector<int>&nums, vector<bool>&used){
        //stop cond
        if(path.size() == nums.size()){
            res.push_back(path);
            return;
        }
        //for
        for(int i = 0; i<nums.size(); i++){
            //cut, 可以将cut理解为判断used[i],或者理解为没有cut
            if(used[i] == false){
                //logic
                path.push_back(nums[i]);
                used[i] = true;
                //reverse
                backtracking(nums,used);
                //backtrack
                path.pop_back();
                used[i] = false;
            }
        }  
    }
};

class Solution {
private:
    vector<int> path;
    vector<vector<int>> res;

public:
    vector<vector<int>> permuteUnique(vector<int>& nums) {
        vector<bool> used(nums.size(),false);
        sort(nums.begin(),nums.end());
        backtracking(nums,used);
        return res;
    }

    void backtracking(vector<int>& nums, vector<bool>& used){
        // stop cond
        if(path.size() == nums.size()){
            res.push_back(path);
            return;
        }

        // for
        for(int i =0;i<nums.size();i++){
            //cut
            
            // used[i-1] == true/false 都行,不代表不写就行......
            // true是树枝去重,false是树层去重
            if(i>0 && nums[i] == nums[i-1] && !used[i-1])
                continue;
            
            if(!used[i]){
                //logic
                path.push_back(nums[i]);
                used[i] = true;
                //recurse
                backtracking(nums,used);
                //backtrack
                path.pop_back();
                used[i] = false;
            }
        }
    }
};

class Solution {
private:
    string path;//声明为全局变量不是一个好主意,传引用不错
    vector<string> res;

public:
    vector<string> permutation(string s) {
        vector<bool> used(s.size(),false);
        sort(s.begin(),s.end());
        backtracking(s,used);
        return res;
    }

    void backtracking(string& s,vector<bool>& used){
        // stop cond
        if(path.size() == s.size()){
            res.push_back(path);
            return;
        }

        //for
        for(int i=0;i<s.size();i++){
            //cut
            if(i>0 && s[i] == s[i-1] && used[i-1] == false)
                continue;

            //logic
            if(!used[i]){
                path.push_back(s[i]);
                used[i] = true;
            //recurse
            backtracking(s,used);

            //backtrack
            path.pop_back();
            used[i] = false;
            }
        }
    }
};
