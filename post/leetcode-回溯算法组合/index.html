class Solution {
private:
    vector<int> path;
    vector<vector<int>> res;

public:
    vector<vector<int>> subsets(vector<int>& nums) {
        backtracking(nums,0);
        return res;
    }

    void backtracking(vector<int>& nums, int startindex){
        // stop cond 
        // 所有路径都需要加入结果集，不需要条件，或者说数组越界自动结束
        if(true){
            res.push_back(path);
        }

        for(int i = startindex; i < nums.size(); i++){
            path.push_back(nums[i]);    //logic
            backtracking(nums,i+1);     //recurse
            path.pop_back();            //backtrace
        }
    }
};

class Solution {
private:
    vector<int> path;
    vector<vector<int>> res;

public:
    vector<vector<int>> subsetsWithDup(vector<int>& nums) {
        sort(nums.begin(), nums.end()); //去重需要先排序
        backtracking(nums,0);
        return res;
    }

    void backtracking(vector<int>&nums, int startindex){
        // stop cond, 所有选项都符合题意,不需要条件,或者说跳出条件是for循环自然结束
        if(true){
            res.push_back(path);
        }
        // for
        for(int i = startindex; i < nums.size(); i++){
            if(i>startindex && nums[i] == nums[i-1])
                continue;               // cut
            path.push_back(nums[i]);    // logic
            backtracking(nums,i+1);     // recurse
            path.pop_back();            // backtrack
        }
    }
};

class Solution {
private:
    vector<int> path;
    vector<vector<int>> res;

public:
    vector<vector<int>> combine(int n, int k) {
        backtracking(n,k,1);
        return res;
    }

    void backtracking(int n, int k, int startindex){
        //stop cond
        if(path.size() == k){
            res.push_back(path);
            return;
        }

        //for
        for(int i=startindex; i<=n; i++){
            //cut 
            //最后元素会因为5超过4越界，函数结束，然后pop出4进而结束，不会进入最后结果集
            
            //logic
            path.push_back(i);
            //recurse
            backtracking(n,k,i+1);
            //backtrace
            path.pop_back();
        }
    }
};

class Solution {
private:
    vector<int> path;
    vector<vector<int>> res;

public:
    vector<vector<int>> combinationSum(vector<int>& candidates, int target) {
        backtraceing(candidates,target,0,0);
        return res;
    }

    void backtraceing(vector<int>& candidates, int target, int startindex, int sum){
        // stop cond
        if(target == sum){
            res.push_back(path);
            return;
        }

        //for
        for(int i = startindex; i< candidates.size();i++){
            //cut
            if(sum > target)
                continue; 

            // handle data
            path.push_back(candidates[i]);
            sum += candidates[i];
            
            // recurse
            backtraceing(candidates, target, i, sum);
            
            // backtracking
            path.pop_back();
            sum -= candidates[i];
        }
    }
};
